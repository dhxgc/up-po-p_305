#include <stdio.h>
int main() {
int n,maximum,minimum; //объявление переменных
scanf("%i",&n); //ввод числа
minimum = 9; //первоначальное значение наименьшей цифры для последующего сравнения
maximum = 0; //первоначальное значение наибольшей цифры для последующего сравнения
while (n>0){ //цикл для поиска наибольшей и наименьшей цифр в числе
if (n%10<minimum) //если остаток от деления (младший разряд) меньше наименьшей цифры,
minimum = n%10; //записываем цифру как наименьшую
if (n%10>maximum) //если остаток от деления (младший разряд) больше наибольшей цифры,
maximum = n%10; //записываем цифру как наибольшую
n = n/10; //переходим к следующей цифре числа
}
printf("%i%i",minimum,maximum); //вывод минимальной цифры и максимальной цифры
}

#include <stdio.h>
int main(){
int n, m, k = 0, res = 0, sum = 0; //объявление переменных
scanf("%i%i",&n,&m); //ввод количества строк и столбцов
int a[n][m]; //объявление двумерного массива
//циклы для ввода элементов двумерного массива
for(int i = 0; i < n; i++){ //цикл для прохода по строкам
for(int j = 0; j < m; j++){ //цикл для прохода по столбцам
scanf("%i",&a[i][j]); //ввод элементов массива
if(a[i][j]>res) //поиск максимального числа в массиве
res = a[i][j]; //запись максимального числа в массиве
}
}
//циклы для поиска количества участников с наилучшим результатом
for(int i = 0; i < n; i++){ //цикл для прохода по строкам
for(int j = 0; j < m; j++){ //цикл для прохода по столбцам
if(a[i][j] == res){ //если элемент двумерного массива имеет наибольшее значение(наилучший результат), то
k++; //прибавляем счетчик количества участников с наилучшим результатом
break; //досрочный выход из цикла
}
}
}
printf("%i",k); //вывод количества участников с наилучшим результатом
}
